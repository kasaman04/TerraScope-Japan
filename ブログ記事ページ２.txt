    function displayCityInfo() {
      const select = document.getElementById("city-select");
      const cityKey = select.value;
      const infoBox = document.getElementById("city-info");
      if (!cityKey || !cityData[cityKey]) {
        infoBox.style.display = "none";
        return;

      const city = cityData[cityKey];
      document.getElementById("city-title").textContent = city.title;
      document.getElementById("population").textContent = city.population;
      document.getElementById("features").textContent = city.features;

      const linksContainer = document.getElementById("article-links");
      linksContainer.innerHTML = "";
      city.articles.forEach(article => {
        const li = document.createElement("li");
        const a = document.createElement("a");
        a.href = article.url;
        a.textContent = article.title;
        li.appendChild(a);
        linksContainer.appendChild(li);
      });

      infoBox.style.display = "block";
